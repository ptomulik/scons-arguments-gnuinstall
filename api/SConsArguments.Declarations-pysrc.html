634                        ('BAR', 'BAR variable help', 'VAR default BAR'),  # VAR 
635                        ('--bar', {'dest':"opt_bar", "type":"string"}))   # OPT 
636          ) 
637   
638      You may of course append keyword arguments to normal arguments to pass 
639      extra declarations. 
640   
641      **Example** 
642   
643      .. python:: 
644   
645          decls = ArgumentDeclarations( 
646             # Argument 'foo' 
647             [('foo',(   {'ENV_FOO' : 'ENV default FOO'},                 # ENV 
648                        ('FOO',         'FOO variable help', ),           # VAR 
649                        ('--foo',       {'dest' : "opt_foo"})         ))],# OPT 
650             # Argument 'geez' 
651             geez  = (   {'ENV_GEEZ' : None},                             # ENV 
652                        ('GEEZ', 'GEEZ variable help', 'VAR default GEEZ'),# VAR 
653                        ('--geez', {'dest':"opt_geez", "type":"string"})) # OPT 
654          ) 
655   
656      or 
657   
658      **Example** 
659   
660      .. python:: 
661   
662          decls = ArgumentDeclarations( 
663             # Argument 'bar' 
664             {'bar':(   {'ENV_BAR' : None},                               # ENV

669                        ('GEEZ', 'GEEZ variable help', 'VAR default GEEZ'),# VAR 
670                        ('--geez', {'dest':"opt_geez", "type":"string"})) # OPT 
671          ) 
672   
673      This function 
674   
675      :Returns: 
676   
677          - `ArgumentDeclarations()` returns an empty `_ArgumentDeclarations` dictionary ``{}``, 
678          - `ArgumentDeclarations(mapping)` returns `_ArgumentDeclarations` dictionary initialized 
679            with ``{k : ArgumentDeclaration(mapping[k]) for k in mapping.keys()}`` 
680          - `ArgumentDeclarations(iterable)` returns `_ArgumentDeclarations` dictionary initialized 
681            with ``{k : ArgumentDeclaration(v) for (k,v) in iter(initializer)}`` 
682   
683          In any case, the keyword arguments ``**kw`` are appended to the 
684          initializer. 
685   
686      """ 
687      convert = lambda x : x if isinstance(x, _ArgumentDeclaration)  \ 
688                             else ArgumentDeclaration(**x) if hasattr(x, 'keys') \ 
689                             else ArgumentDeclaration(*tuple(x)) 
690      return _ArgumentDeclarations(__dict_converted(convert, *args, **kw))
